类和对象-构造函数和析构函数

发表于 2016-03-30   |  

构造函数:具有构造函数的类在每次创建对象的时候会先调用构造函数

<?php
//子类自动继承父类构造函数
class A{
    public function __construct(){
        echo get_class()."</br>";
    }
}
class B extends A{

}
new A();//输出 A
new B();//输出 A

子类声明构造函数后会覆盖父类的构造函数

<?php
class A{
    public function __construct(){
        echo get_class()."</br>";
    }
}
class B extends A{
    public function __construct(){
        echo get_class();
    }
}
new A();//输出 A
new B();//输出 B

子类可以在构造函数中执行父类的构造函数

<?php
class A{
    public function __construct(){
        echo get_class()."</br>";
    }
}
class B extends A{
    public function __construct(){
        parent::__construct();
        echo get_class();
    }
}
new A();//输出 A
new B();//输出 A B

子类构造函数覆盖父类构造函数时

<?php
class A{
    public function __construct($a){
        echo $a."</br>";
    }
}
class B extends A{
    public function __construct($a,$b){
        // parent::__construct();
        //这行要是放开注释会报一个 warning 和 notice 因为子类构造函数和父类构造函数的参数不同
        echo $a.$b."</br>";
    }
}
new A('a');//输出 a
new B('a','b');//输出 a b

在没有命名空间时,在还没有找到构造函数时,会把与类同名的函数当做构造函数

<?php
class test{
    public function __construct(){
        echo 'construct'."</br>";
    }
    public function test(){
        echo 'test'."</br>";
    }
}
class test1{
    public function test1(){
        echo 'test'."</br>";
    }
    public function __construct(){
        echo 'construct'."</br>";
    }
}
class test2{
    public function test2(){
        echo 'test2';
    }
}
new test();//construct
new test1();//construct,报一个严格的错误检查:提示构造函数被覆
new test2();//test2

有命名空间时,和类同名的函数,不会吧同名方法当做构造函数

<?php
namespace Test;
class Test{
    public function Test(){
        echo test;
    }
    public function __construct(){
        echo 'test';
    }
}
class Test1{
    public function __construct(){
        echo 'test1';
    }
    public function Test1(){
        echo 'test1';
    }
}
new Test();//test
new Test1();//test1

私有构造函数

<?php
    class Test{
        const abc='abc';
        public $a='a';
        static $b='b';
        private function __construct(){
            echo 'test';
        }
        public function geta(){
            return 'a';
        }
        static function getb(){
            return 'b';
        }
    }
    echo Test::abc;//abc
    echo Test::$b;//b
    echo Test::getb();//b
    echo Test::geta();//a 并报 strict standards 因为用静态方法调用非静态方法
    class Test2 extends Test{

    }
    // new Test();//声明一个构造函数是私有的类会报 fatalerror 
    // new Test2();//fatalerror 和上面一样
    class Test3 extends Test{
        public function __construct(){
            echo 'test3';
        }
    }
    $a = new Test3();//test3可以正常实例化

析构函数:析构函数在对象被销毁时调用 

<?php
//test.php
class Test{
    public function  __destruct(){
        echo 'this is  test desstruct'."</br>";
    }
}
class Test1 extends Test{

}
class Test2 extends Test{
    public function __destruct(){
        parent::__destruct();
        echo 'this is test2 desstruct'."</br>";

    }
}

new Test();//this is test desstruct
new Test1();//this is test desstruct
new Test2();//this is test desstruct,this is test2 desstruct
require_once('test2.php');
new test3();//this is test2 file ,脚本关闭时的文件和被请求的文件不是一个

//text2.php
<?php
class test3{
    public function __destruct(){
        echo 'this is test2 file'."</br>";
    }
}

exit() 和 die() 时候也会 调用析构函数

<?php
class Test{
    public function __construct(){
        echo 'Start'."</br>";
        die();
    }

    public function  __destruct(){
        echo 'exit will exec destruct'."</br>";
    }
}
new Test();
//输出:
// Start
// exit will exec destruct

第一个构造函数的 demo

    <?php
class test{
    public function __construct(){
        $argArr = func_get_args();
        $num = func_num_args();
        if(method_exists($this, $name='arg'.$num)){
            call_user_func_array("self::$name", $argArr);
        }
        else{
            echo 'args illegal';
        }
    }
    public static function arg1($a){
        echo $a."</br>";
    }
    public static function arg2($a,$b){
        echo $a.' '.$b."</br>";
    }
    public static function arg3($a,$b,$c){
        echo $a.' '.$b.' '.$c."</br>";
    }
}
new test('a');//a
new test('a','b');//a b
new test('a','b','c');//a b c

当两个对象互相引用时,析构函数被执行的时机就很诡异

<?php
header('Content-type:text/plain');
class Foo{
    private $name;
    private $link;
    public function __construct($name){
        $this->name = $name;
    }
    public function setLink(Foo $link){
        $this->link = $link;
        print_r($link);
    }
    public function __destruct(){
        echo 'destruct:',$this->name, PHP_EOL;
    }
}
$foo = new Foo('Foo 1');
$bar = new Foo('Foo 2');

$foo->setLink($bar);
$bar->setLink($foo);
$foo = null;
$bar = null;

$foo = new Foo('Foo 3');
$bar = new Foo('Foo 4');

$foo = null;
$bar = null;
echo 'End of script', PHP_EOL;
//会输出这些
//     Foo Object
// (
//     [name:Foo:private] => Foo 2
//     [link:Foo:private] => 
// )
// Foo Object
// (
//     [name:Foo:private] => Foo 1
//     [link:Foo:private] => Foo Object
//         (
//             [name:Foo:private] => Foo 2
//             [link:Foo:private] => Foo Object
//  *RECURSION*
//         )

// )
// destruct:Foo 3
// destruct:Foo 4
// End of script
// destruct:Foo 1
// destruct:Foo 2
