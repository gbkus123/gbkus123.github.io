类和对象-类常量

发表于 2016-03-27   |  

定义和调用类常量

类中的常量是始终保持不变的值,在定义时不需要$符号,常量的值必须是一个定值,不能使变量,类属性,数学运算的结果(在php5.6之后已经可以是数学运算的结果了)),或者函数返回的结果,这一点和类的属性很像,定义常量用const关键字

    <?php
//php5的某个版本之后 const 既可以在类中也可以在类外定义常量
const A='1'."</br>";
echo A;//1,在类外的常量可以直接获取
const STR='out of class'.'</br>';
class test{
    const STR='inside class'.'</br>';
    function getCon(){
        echo self::STR;//inside class获取在类中的常量用 self::
        echo STR;//out of class不用 self::会去类外找该常量
        echo NOCONST;//NOCONST 如果试图去输出一个没有定义的常量,会报 Notice,并输出该常量的名字
        echo "</br>";
    }
}
class child extends test{
    function getCon(){
        echo self::STR;//inside class
        echo parent::STR;//inside class 子类这里也可以用 parent::
        echo STR;//out of class 如果不用 self::会去类外找该常量
        echo NOCONST;//如果试图去输出一个没有定义的常量,会报 Notice,并输出该常量的名字
        echo "</br>";
    }
}
$t = new test();
$t->getCon();
$c = new child();
$c->getCon();
//直接获取类的常量,获取用变量存储类名,或者用实例化的对象来获取常量都是可以的
echo test::STR;//inside class
$class='test';//inside class
echo $class::STR;//inside class
echo $t::STR;//inside class
echo $c::STR;//inside class

doc 格式与常量

<?php
class test {
    const heredoc=<<<HERE
    here
HERE;
    const newdoc=<<<'new'
    new
new;
}
echo test :: heredoc;//here
echo test ::newdoc;//new

<?php    
class dbObject{
    const TABLE_NAME='none';
    static public function  getQuery(){
        $c=get_called_class();//返回调用该静态方法的类
        return "SELECT * FROM `".$c::TABLE_NAME."`";
    }
}
class tableUser extends dbObject{
    const TABLE_NAME='user';
}
class tableOrder extends dbObject{
    const TALBE_NAME='order';
}
echo dbObject::getQuery()."</br>";
echo tableUser::getQuery()."</br>";
echo tableOrder::getQuery()."</br>";
//SELECT * FROM `none`
//SELECT * FROM `user`

<?php
//self::,static:: +常量,和 new self new static 一样
//self 返回代码所在类的常量, static返回调用该函数的类的常量
class A{
    const BOOL = 'A';
    public function getSelf(){
        return self::BOOL;
    }
    public function getStatic(){
        return static::BOOL;
    }
}
class B extends A{
    const BOOL = 'B';
}
$b=new B();
echo $b->getSelf()."</br>";//A
echo $b->getStatic();//B
