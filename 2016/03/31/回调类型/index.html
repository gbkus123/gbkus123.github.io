回调类型

发表于 2016-03-31   |  

回调类型:一些函数 可以接受用户自定义的函数 对象的方法(包括静态方法)作为参数

传递

php 函数在传递时 函数名以 string 传递,包括内置和用户自定义函数,但不包括语言结构

实例化后的对象,例如:$a=new a()在传递时,要传递一个[0=>$a,1=>方法名]的数组,

静态的方法可以不经过实例化该类,有两种方法:

1:数组[0=>类名,1=>方法名+]]

2:一个参数 class::functionname

 <?php 
class B{
    public function __construct(){
        $name=func_get_args();
        $i=func_num_args();
        if(method_exists($this,'arguments'.$i)){
            call_user_func_array([$this,'arguments'.$i],$name);
        }else{
            call_user_func('self::noFunc','not have this func');
        }
    }
    static public function noFunc($a){
        echo $a;
    }
    public function arguments1($a){
        echo $a;
    }
    public function arguments2($a,$b){
        echo $a.$b;
    }
    public function arguments3($a,$b,$c){
        echo $a.$b.$c;
    }
}
new B('a','b','c','d');//输出 not have this func
new B('a');//输出 a
new B('a','b');//输出 ab 
new B('a','b','c');//输出 abc

各种传递方式:

 <?php 
function test(){
    echo 'this is test func'."</br>";
}
class  myclass{
    public function my1($a){
        echo $a."</br>";
    }
    static public function my2($a,$b){
        echo $a.$b."</br>";
    }
}
class other extends  myclass{
    static public function my2($a,$b){
        echo 'this is other my2';
    }
}

//不在类中的函数,直接写方法名字
call_user_func('test');//this is test func 

$m = new myclass();
$n = new other();

//第一种方式第一个参数是数组对象和方法名,后面加参数
call_user_func([$m,'my1'],'this is my1 func');//this is my1 func

//静态方法第一个数组参数是类名和方法名
call_user_func_array(['myclass','my2'],['this is ','my2 func']);//this is my2 func

//静态方法可以写作正常调用静态方法的方式来写
call_user_func_array('myclass::my2',['this is ','my2 func']);//this is my2 func 

//子类可以正常调用父类费静态方法
call_user_func([$n,'my1'],'this is other func');//this is other func 

//子类亦可以用调用父类静态方法的方式来写
call_user_func_array(['other','parent::my2'],['this is ','other func']);//this is other func 

//这种方式是会报错的 ,我以为会有效
// call_user_func_array(['other::my2'],['this is ','my2 func']);

如果一个回调函数有异常没有被 catch ,并且在这个回调函数的后面还有别的回调,那么后面的回调函数不会被执行

<?php
function func1(){
    throw new Exception('this is exception');

}
function func2(){
    echo 'this is func2';
}
class test{
    public function __construct(){
        try{
            call_user_func('func1');
        }catch(Exception $e){
            echo $e->getMessage();
        }
        call_user_func('func2');
    }
}
new test();

一个绕来绕去的 demo

<?php
class A{
    static public function fromHere(){
        B::crossHere(self::doHere);
    }
    static public function doHere(){
        echo 'success';
    }
}
class B{
    static public function crossHere(callable $callName){
        call_user_func($callName);
    }
}
A::fromHere();
// Fatal error: Undefined class constant 'doHere'
//找不到 doHere 方法,就是说 self::doHere 并不是先找到 doHere 方法然后传过去 
//而是先传过去再去找,self::doHere,并且如果被self::doHere被字符串包起来会报不是回调类型的错误    
//将第四行改为B::crossHere('A::doHere');B::crossHere(['A','doHere']); 会正常执行 doHere

私有函数可以在类内作为回调函数,在类外不行 可是这又有什么用呢,谁不知道啊 ,不知道为什么在官网还有180个赞同

<?php
class test{
    private function arrayWalk($val,$k,$userData){
        echo $userData.$k.'=>'.$val.'</br>';
    }
    public function inside(){
        $arr = ['a','b','c','d','e','f'];
        array_walk($arr,[$this,'arrayWalk'],'this is  ');
    }
    public function outside(){
        return [$this,'arrayWalk'];
    }
}
$t = new test();
$t->inside();
//输出this is 0=>a
//this is 1=>b
//this is 2=>c
//this is 3=>d
//this is 4=>e
//this is 5=>f
$f = $t->outside();
call_user_func($f);//报错
//Warning: call_user_func() expects parameter 1 to be a valid callback, cannot access private method test::arrayWalk()

class test{

        $c();
    }

    }
    public static function beCalled(){
        echo 'success'."</br>";

test::callFunc2('test::beCalled');//success 和 一个静态方法需要被静态调用
test::callFunc2(['test','beCalled']);//success 和一个静态方法需要被静态调用
