访问控制

发表于 2016-04-07   |  

对属性和方法的访问控制有三个关键字:

public : 公用的类成员可以在任何地方被访问

protected : 受保护的类成员可以被自身 子类 父类访问

private : 私有的类成员只能在定义的类里访问

属性的访问控制

<?php

class A{
    public $publicA = 'publicA'."</br>";
    protected $protectedA = 'protectedA'."</br>";
    private $privateA = 'privateA'."</br>";
    public function pulicFuncA(){
        echo 'this is publicA'."</br>";
    }
    protected function protectedFuncA(){
        echo 'this is protectedA'."</br>";
    }
    private function privateFuncA(){
        echo 'this is privateA'."</br>";
    }
    public function getProtected(){
        echo $this->protectedA;
    }
    public function getPrivate(){
        echo $this->privateA ;
    }

}
$a = new A();
echo $a->publicA;//publicA
// echo $a->protectedA;//fatal error
// echo $a->privateA;//fatal error
$a->pulicFuncA();//this is publicA
// $a->protectedFuncA();//fatal error
// $a->privateFuncA();//fatal error
$a->getProtected();//protectedA
$a->getPrivate();//privateA

//子类可以修改父类Public protected 的属性
//同名的 private 的属性不可以被重置,子类调用父类的 '获取同名私有属性' 方法时,
//访问的是父类的 私有属性,下面这个例子中获取子类的私有属性,重新写一个
//获取该私有属性的方法
class B extends A{
    public $publicA = 'publicB'."</br>";
    protected $protectedA = 'protectedB'."</br>";
    private $privateA = 'asdf';
    public function getPrivateB(){
        echo $this->privateA;
    }
}
$b = new B();
echo $b->publicA;//publicB
$b->getProtected();//protectedB
$b->getPrivate();//privateA
$b->getPrivateB();//asdf

方法的访问控制

方法的访问控制,声明访问控制和属性一样,如果不声明默认 Public

<?php
class A 
{
    public function b(){
        echo 'this is b '."</br>";
    }
    protected function c(){
        echo 'this is c '."</br>";
    }
    private function d(){
        echo 'this is d '."</br>";
    }
}
$a = new A();

$a->b();//this is b
// $a->c();//fatal error
// $a->d();//fatal error

同属性一样父类的 Pulic protected 方法可以被调用,也可以重载,但 private 方法不可以

<?php
class A 
{
    public function b(){
        echo 'this is A b '."</br>";
    }
    protected function c(){
        echo 'this is A c '."</br>";
    }
    private function d(){
        echo 'this is A d '."</br>";
    }
    public  function execProtected(){
        $this->c();
    }
    public function execPrivate(){
        $this->d();
    }
}
class Bar extends A{
    public function b(){
        echo 'this is B b '."</br>";
    }
    protected function c(){
        echo 'this is B c '."</br>";
    }
    private function d(){
        echo 'this is A d '."</br>";
    }
}
$a = new A();
$b = new Bar();
$b -> b();//this is B b 
$b -> execProtected();//this is B c 
$b -> execPrivate();//this is A d

同一个类的不同实例可以互相访问该类的private protected 成员

<?php
class test{
    private $bar = '';
    protected $foo = '';
    public function __construct($pir,$pro){
        $this->bar = $pir;
        $this->foo = $pro;
    }
    private function abc(){
        echo 'this is private func'."</br>";
    }
    protected function efg(){
        echo 'this is protected func'."</br>";
    }
    public function getProperty(test $obj){

        echo $obj->bar;
        echo $obj->foo;
        $obj->abc();
        $obj->efg();
    }
}
$a = new test('test ','1');
$a->getProperty(new test('test ','2'));
// 输出 
// test 2
// this is private func
//this is protected func

一歪果仁认为简单的设置对象的类属性$newObj->property=1; 和简单的获取类属性是不对的,因为你不知道这个类属性是怎么定义的,你可以给类属性任意的值 ,所以他建议用以下这种写法,其实我觉得多此一举,我觉得别的强类型和纯面向对象语言都是这么干的:

<?php
class test{
    protected $name = 'undefined';
    protected $money = 0.00;

    public function getName(){
        return $this->name;
    }
    public function getMoney(){
        return $this->money;
    }
    public function setName($name){
        if(is_string($name)){
            $this->name = strval($name);
        }
    }
    public function setMoney($money){
        if(is_numeric($money)){
            $this->money = floatval($money);
        }
    }
}
$a = new test();
$a->setName(1);
$a->setMoney('abc');
echo $a->getName();
echo $a->getMoney();
// 输出  undefined 0
$a->setName('lilei');
$a->setMoney(250);
echo $a->getName();
echo $a->getMoney();
// 输出  lilei250

上面这个例子如果不这么写的话 其实是不需要那四个 function 的

可以一次声明多个类属性

    <?php
class test{
    public $a = 1,$b = 2;
    protected $c = 3, $d = 4;
    private $e = 5,$f = 6;
    function getPro(){
        echo $this->c,$this->d;
    }
    function getPri(){
        echo $this->e,$this->f;
    }
}
$a = new test();
echo $a->a,$a->b;
$a->getPro();
$a->getPri();
// 输出  123456

这个例子主要是说明父类不能调用子类的 private 方法,这个东西不需要太复杂就没改直接 copy 的

    <?php

class ParentClass {

    public function execute($method) {
        $this->$method();
    }

}

class ChildClass extends ParentClass {

    private function privateMethod() {
        echo "hi, i'm private";
    }

    protected function protectedMethod() {
        echo "hi, i'm protected";
    }

}

$object = new ChildClass();

$object->execute('protectedMethod');

$object->execute('privateMethod');
//输出
//    hi, i'm protected
//Fatal error: Call to private method ChildClass::privateMethod() from context 'ParentClass' in index.php on line 6
