类和对象-重载

发表于 2016-04-20   |  

php 是动态的创建 类属性和方法,是通过魔术方法实现

当调用当前环境下,未定义或者 不可见的类属性或方式时,从在方法(set,get__call)会被调用,

所有的重载方法必须被声明为 public

重载方法不能通过引用传递

属性的重载

<?php
class test{
    private $arr = [];
    public $declared = 'can\'t overwrite';
    private $hidden = 'this is hidden propery';
    public function __set($name,$value){
        echo "setting $name to $value </br>";
        $this->arr[$name] = $value;
    }
    public function __get($name){
        echo "getting $name </br>";
        if(array_key_exists($name, $this->arr)){
            return $this->arr[$name];
        }
        $trace = debug_backtrace();
        trigger_error("undefined property 
            $name in {$trace[0]['file']} on line {$trace[0]['line']}
            ,E_USER_NOTICE");
        return null;
    }
    public function __isset($name){
        echo "is $name set? </br>";
        return isset($this->arr[$name]);
    }
    public function __unset($name){
        echo "unsetting $name </br>";
        unset($this->arr[$name]);
    }
    public function getHidden(){
        return $this->hidden;
    }
}
$t = new test();

$t->notHave = 'notHave';
echo $t->notHave."</br>";

var_dump(isset($t->notHave));
unset($t->notHave);
var_dump(isset($t->notHave));

echo $t->declared."</br>";

echo $t->getHidden();

echo $t->hidden;
//输出
// setting notHave to notHave 
// getting notHave 
// notHave
// is notHave set? 
// boolean true
// unsetting notHave 
// is notHave set? 
// boolean false
// can't overwrite
// this is hidden properygetting hidden 
// Notice: undefined property hidden in /Users/guanbingqiu/www/test.php on line 46 
// ,E_USER_NOTICE in /Users/name/www/test.php on line 18

范围解析操作符`::`

发表于 2016-04-19   |  

范围解析操作符 ,就是两个冒号

可以用来访问静态成员 类常量

还可以用于覆盖类中的属性和方法

在类外使用::时要在前面 加上类名

<?php
class Myclass{
    const Myconst = 'this is myclass \'s const';
}
class Test extends MyClass{
    const testConst = 'this is test const';
    public static function  testFunc(){
        return 'this is test func';
    }
    public static function getConst(){
        return static::testConst;
        //此处 static 引用全部静态域,并且子类的静态成员会覆盖父类的静态成员
    }
    public static function getConstInself(){
        return self::testConst;
    }
    public static function getParentConst(){
        return  parent::Myconst;
    }
}
$t = new Test();
$name = 'Test';
echo $name::Myconst."</br>";//this is myclass's const
echo Test::testConst."</br>";//this is test const
echo Test::testFunc()."</br>";//this is test func
echo Test::getConst()."</br>";//this is test const
echo Test::getConstInself()."</br>";//this is test const
echo Test::getParentConst()."</br>";//this is myclass 's const

当子类覆盖父类中的方法时, php 不会调用父类的方法,如果过此时还需要调用父类的方法时,可以用 parent::来实现 ,这种机制包括构造,析构函数,重载和魔术方法

构造和析构函数

<?php
class theParent{
    public function __construct(){
        echo 'this is parent\'s  constucrt'."</br>";
    }
    public function myFunc(){
        echo 'parent\'s myfunc'."</br>";
    }

    public function __destruct(){
        echo 'this is parent\'s  constucrt'."</br>";
    }
}
class child extends theParent{
    public function __construct(){
        parent::__construct();
        echo 'this is child\'s  constucrt'."</br>";
    }
    public function myFunc(){
        parent::myFunc();
        echo 'child\'s myfunc'."</br>";
    }
    public function __destruct(){
        parent::__construct();
        echo 'this is child\'s  constucrt'."</br>";
    }
}
$c = new child();
$c->myfunc();
//输出:
// this is parent's constucrt
// this is child's constucrt
// parent's myfunc
// child's myfunc
// this is parent's constucrt
// this is child's constucrt

访问控制

发表于 2016-04-07   |  

对属性和方法的访问控制有三个关键字:

public : 公用的类成员可以在任何地方被访问

protected : 受保护的类成员可以被自身 子类 父类访问

private : 私有的类成员只能在定义的类里访问

属性的访问控制

<?php

class A{
    public $publicA = 'publicA'."</br>";
    protected $protectedA = 'protectedA'."</br>";
    private $privateA = 'privateA'."</br>";
    public function pulicFuncA(){
        echo 'this is publicA'."</br>";
    }
    protected function protectedFuncA(){
        echo 'this is protectedA'."</br>";
    }
    private function privateFuncA(){
        echo 'this is privateA'."</br>";
    }
    public function getProtected(){
        echo $this->protectedA;
    }
    public function getPrivate(){
        echo $this->privateA ;
    }

}
$a = new A();
echo $a->publicA;//publicA
// echo $a->protectedA;//fatal error
// echo $a->privateA;//fatal error
$a->pulicFuncA();//this is publicA

定义和调用类常量

类中的常量是始终保持不变的值,在定义时不需要$符号,常量的值必须是一个定值,不能使变量,类属性,数学运算的结果(在php5.6之后已经可以是数学运算的结果了)),或者函数返回的结果,这一点和类的属性很像,定义常量用const关键字

    <?php
//php5的某个版本之后 const 既可以在类中也可以在类外定义常量
const A='1'."</br>";
echo A;//1,在类外的常量可以直接获取
const STR='out of class'.'</br>';
class test{
    const STR='inside class'.'</br>';
    function getCon(){
        echo self::STR;//inside class获取在类中的常量用 self::
        echo STR;//out of class不用 self::会去类外找该常量
        echo NOCONST;//NOCONST 如果试图去输出一个没有定义的常量,会报 Notice,并输出该常量的名字
        echo "</br>";
    }
}
class child extends test{
    function getCon(){
        echo self::STR;//inside class
        echo parent::STR;//inside class 子类这里也可以用 parent::
        echo STR;//out of class 如果不用 self::会去类外找该常量
        echo NOCONST;//如果试图去输出一个没有定义的常量,会报 Notice,并输出该常量的名字
        echo "</br>";
    }
}
$t = new test();
$t->getCon();
$c = new child();
$c->getCon();
//直接获取类的常量,获取用变量存储类名,或者用实例化的对象来获取常量都是可以的
echo test::STR;//inside class
$class='test';//inside class
echo $class::STR;//inside class
echo $t::STR;//inside class
echo $c::STR;//inside class
