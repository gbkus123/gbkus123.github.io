类和对象-重载

发表于 2018-11-22   |  

php 是动态的创建 类属性和方法,是通过魔术方法实现

当调用当前环境下,未定义或者 不可见的类属性或方法时,重载方法(set,get__call)会被调用,

所有的重载方法必须被声明为 public

重载方法不能通过引用传递

属性的重载

<?php
class test{
    private $arr = [];
    public $declared = 'can\'t overwrite';
    private $hidden = 'this is hidden propery';
    public function __set($name,$value){
        echo "setting $name to $value </br>";
        $this->arr[$name] = $value;
    }
    public function __get($name){
        echo "getting $name </br>"; 
        if(array_key_exists($name, $this->arr)){
            return $this->arr[$name];
        }
        $trace = debug_backtrace();
        trigger_error("undefined property 
            $name in {$trace[0]['file']} on line {$trace[0]['line']}
            ,E_USER_NOTICE");
        return null;
    }
    public function __isset($name){
        echo "is $name set? </br>";
        return isset($this->arr[$name]);
    }
    // public function __unset($name){
    //     echo "unsetting $name </br>";
    //     unset($this->arr[$name]);
    // }
    public function getHidden(){
        return $this->hidden;
    }
}
$t = new test();

$t->notHave = 'notHave';
echo $t->notHave."</br>";

var_dump(isset($t->notHave));
unset($t->notHave);
var_dump(isset($t->notHave));

echo $t->declared."</br>";

echo $t->getHidden();

echo $t->hidden;

//输出

// setting notHave to notHave 
// getting notHave 
// notHave
// is notHave set? 
// boolean true
// unsetting notHave 
// is notHave set? 
// boolean false
// can't overwrite
// this is hidden properygetting hidden 
// Notice: undefined property hidden in /Users/guanbingqiu/www/test.php on line 46 
// ,E_USER_NOTICE in /Users/name/www/test.php on line 18

$t->otherNotHave = 'the other not have';
var_dump(isset($t->declared));//true
var_dump($t->otherNotHave);//the other not have 
var_dump(isset($t->otherNotHave));//true 
$t->otherNotHave = 'change the other not have';
var_dump($t->otherNotHave);
$t->declared = 'change can\'t overwrite';
var_dump($t->declared);//change can\'t overwrite'
$t->declared = 1;
var_dump($t->declared);//1
unset($t->declared);
var_dump($t->declared);//报错: notice undifined property declared 并输出:null
var_dump(isset($t->declared));//false

方法的重载,call,如果是静态的调用就用callStatic

n<?php
class test{
    public function __call($name,$arguments){
        $this->$name($arguments);
    }    
    public static function __callStatic($name,$arguments){
        static::staticFunc($name,$arguments);
    }
    private function func($arg){
        var_dump($arg);
    }
    private static function staticFunc($name,$arg){
        var_dump($name,$arg);
    }
}
$t = new test();

$t->func('a','b');
test::staticFunc('c','d');
// array (size=2)
//   0 => string 'a' (length=1)
//   1 => string 'b' (length=1)
// string 'staticFunc' (length=10)
// array (size=2)
//   0 => string 'c' (length=1)
//   1 => string 'd' (length=1)
